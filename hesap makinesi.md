        private void esittirbutton_Click(object sender, EventArgs e)
        {
            int ikinisayi = Convert.ToInt32(label1.Text);
            int sonuç;



            switch(_islem)
            {
                case '+':
                    sonuç = _ilksayi + ikinisayi;
                    break;
                case '-':
                    sonuç = _ilksayi - ikinisayi;
                    break;
                case '*':
                    sonuç = _ilksayi * ikinisayi;
                    break;
                case '/':
                    sonuç = _ilksayi / ikinisayi;
                    break;
                default:
                    sonuç = 0;
                    break;
            }

            label1.Text = Convert.ToString(sonuç);

        }
